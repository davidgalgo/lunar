// --- Gradual Blur Class (Refactored to remove math.js dependency) ---

class GradualBlur {

    constructor(targetElement, config = {}) {

        this.containerRef = null;

        this.isHovered = false;


        const DEFAULT_CONFIG = {

            position: 'bottom',

            strength: 2,

            height: '6rem',

            divCount: 5,

            exponential: false,

            zIndex: 1000,

            opacity: 1,

            curve: 'linear',

            target: 'parent',

            className: '',

            style: {}

        };


        this.config = { ...DEFAULT_CONFIG, ...config };

        this.targetElement = targetElement;


        this.CURVE_FUNCTIONS = {

            linear: p => p,

            bezier: p => p * p * (3 - 2 * p),

            'ease-in': p => p * p,

            'ease-out': p => 1 - Math.pow(1 - p, 2),

            'ease-in-out': p => (p < 0.5 ? 2 * p * p : 1 - Math.pow(-2 * p + 2, 2) / 2)

        };

        

        this.init();

    }

    

    getGradientDirection(position) {

        const directions = {

            top: 'to top',

            bottom: 'to bottom',

            left: 'to left',

            right: 'to right'

        };

        return directions[position] || 'to bottom';

    }


    createBlurDivs() {

        const divs = document.createDocumentFragment();

        const increment = 100 / this.config.divCount;

        const currentStrength = this.isHovered && this.config.hoverIntensity ? this.config.strength * this.config.hoverIntensity : this.config.strength;

        const curveFunc = this.CURVE_FUNCTIONS[this.config.curve] || this.CURVE_FUNCTIONS.linear;


        for (let i = 1; i <= this.config.divCount; i++) {

            let progress = i / this.config.divCount;

            progress = curveFunc(progress);


            let blurValue;

            if (this.config.exponential) {

                // Replaced math.pow with Math.pow

                blurValue = Math.pow(2, progress * 4) * 0.0625 * currentStrength;

            } else {

                blurValue = 0.0625 * (progress * this.config.divCount + 1) * currentStrength;

            }

            

            // Replaced math.round with Math.round

            const p1 = Math.round((increment * i - increment) * 10) / 10;

            const p2 = Math.round(increment * i * 10) / 10;

            const p3 = Math.round((increment * i + increment) * 10) / 10;

            const p4 = Math.round((increment * i + increment * 2) * 10) / 10;

            let gradient = `transparent ${p1}%, black ${p2}%`;

            if (p3 <= 100) gradient += `, black ${p3}%`;

            if (p4 <= 100) gradient += `, transparent ${p4}%`;

            

            const direction = this.getGradientDirection(this.config.position);


            const div = document.createElement('div');

            div.style.position = 'absolute';

            div.style.inset = '0';

            div.style.maskImage = `linear-gradient(${direction}, ${gradient})`;

            div.style.webkitMaskImage = `linear-gradient(${direction}, ${gradient})`;

            div.style.backdropFilter = `blur(${blurValue.toFixed(3)}rem)`;

            div.style.webkitBackdropFilter = `blur(${blurValue.toFixed(3)}rem)`;

            div.style.opacity = this.config.opacity;


            divs.appendChild(div);

        }

        return divs;

    }

    

    applyContainerStyles() {

        const { position, target, hoverIntensity, zIndex, style, height, width } = this.config;

        const isVertical = ['top', 'bottom'].includes(position);

        const isHorizontal = ['left', 'right'].includes(position);

        const isPageTarget = target === 'page';


        const baseStyle = {

            position: isPageTarget ? 'fixed' : 'absolute',

            pointerEvents: hoverIntensity ? 'auto' : 'none',

            zIndex: isPageTarget ? 30 : zIndex, // Modified z-index for page target to be below header/modal

            ...style

        };

        

        Object.assign(this.containerRef.style, baseStyle);

        

        if (isVertical) {

            this.containerRef.style.height = height;

            this.containerRef.style.width = width || '100%';

            this.containerRef.style[position] = '0';

            this.containerRef.style.left = '0';

            this.containerRef.style.right = '0';

        } else if (isHorizontal) {

            this.containerRef.style.width = width || height;

            this.containerRef.style.height = '100%';

            this.containerRef.style[position] = '0';

            this.containerRef.style.top = '0';

            this.containerRef.style.bottom = '0';

        }

    }


    init() {

        this.containerRef = document.createElement('div');

        this.containerRef.className = `gradual-blur ${this.config.className}`;

        

        this.applyContainerStyles();


        const innerDiv = document.createElement('div');

        innerDiv.className = 'gradual-blur-inner';

        innerDiv.style.position = 'relative';

        innerDiv.style.width = '100%';

        innerDiv.style.height = '100%';

        

        innerDiv.appendChild(this.createBlurDivs());

        this.containerRef.appendChild(innerDiv);

        

        if (this.config.hoverIntensity) {

            this.containerRef.addEventListener('mouseenter', () => {

                this.isHovered = true;

                innerDiv.innerHTML = '';

                innerDiv.appendChild(this.createBlurDivs());

            });

             this.containerRef.addEventListener('mouseleave', () => {

                this.isHovered = false;

                innerDiv.innerHTML = '';

                innerDiv.appendChild(this.createBlurDivs());

            });

        }

        

        this.targetElement.appendChild(this.containerRef);

    }

}
